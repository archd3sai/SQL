SELECT F.ACTIVITY FROM FRIENDS F
GROUP BY F.ACTIVITY
HAVING COUNT(*) 
NOT IN(
(SELECT MAX(TOTAL) FROM (SELECT COUNT(*) AS TOTAL
FROM FRIENDS GROUP BY ACTIVITY) AS T1),

(SELECT MIN(TOTAL) FROM (SELECT COUNT(*) AS TOTAL
FROM FRIENDS GROUP BY ACTIVITY) AS T2)
)
